// media queries

@mixin MQ($canvas) {

   @media only screen and (min-width: $S) { @content; } 
  }

   @media only screen and (min-width: $M) { @content; } 
  }
  @else if $canvas == L {
   @media only screen and (min-width: $L) { @content; } 
  }
}
